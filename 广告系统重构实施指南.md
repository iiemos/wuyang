#### Step 2: 创建 Service 类

创建 `server/src/modules/operation/ad-campaign.service.ts`：

```typescript
import { Injectable } from '@nestjs/common'
import { PrismaService } from '../prisma/prisma.service'
import { z } from 'zod'

const campaignSchema = z.object({
  name: z.string().min(1).max(128),
  description: z.string().optional(),
  startDate: z.string().refine(d => !isNaN(Date.parse(d))),
  endDate: z.string().refine(d => !isNaN(Date.parse(d))),
  budget: z.number().optional().nullable(),
  dailyBudget: z.number().optional().nullable(),
  pricingModel: z.enum(['cpm', 'cpc', 'cpa']).default('cpm'),
  unitPrice: z.number().optional().nullable(),
  positionIds: z.array(z.string()),
  timeSlots: z.array(z.object({
    start: z.string(),
    end: z.string()
  })).optional(),
  advertiserName: z.string().optional(),
  contactPhone: z.string().optional(),
  contactEmail: z.string().optional()
})

@Injectable()
export class AdCampaignService {
  constructor(private readonly prisma: PrismaService) {}

  // 创建计划
  async createCampaign(data: unknown, operator: string) {
    const validated = campaignSchema.parse(data)

    return this.prisma.adCampaign.create({
      data: {
        ...validated,
        startDate: new Date(validated.startDate),
        endDate: new Date(validated.endDate),
        positionIds: JSON.stringify(validated.positionIds),
        timeSlots: validated.timeSlots ? JSON.stringify(validated.timeSlots) : null,
        status: 'draft'
      }
    })
  }

  // 获取计划列表
  async listCampaigns(query: {
    page?: number
    pageSize?: number
    status?: string
    keyword?: string
  }) {
    const page = query.page || 1
    const pageSize = Math.min(query.pageSize || 20, 100)
    const skip = (page - 1) * pageSize

    const where = {
      ...(query.status && { status: query.status }),
      ...(query.keyword && {
        OR: [
          { name: { contains: query.keyword } },
          { advertiserName: { contains: query.keyword } }
        ]
      })
    }

    const [items, total] = await Promise.all([
      this.prisma.adCampaign.findMany({
        where,
        skip,
        take: pageSize,
        orderBy: { createdAt: 'desc' },
        include: {
          creatives: {
            select: { id: true, title: true, status: true }
          }
        }
      }),
      this.prisma.adCampaign.count({ where })
    ])

    return {
      items,
      total,
      page,
      pageSize
    }
  }

  // 获取计划详情
  async getCampaignDetail(id: string) {
    return this.prisma.adCampaign.findUnique({
      where: { id },
      include: {
        creatives: {
          select: {
            id: true,
            title: true,
            image: true,
            status: true,
            impression: true,
            click: true,
            reviewedAt: true
          }
        }
      }
    })
  }

  // 更新计划状态
  async updateCampaignStatus(id: string, status: string, rejectReason?: string) {
    return this.prisma.adCampaign.update({
      where: { id },
      data: {
        status,
        rejectReason: rejectReason || null
      }
    })
  }

  // 更新日限额
  async updateDailyBudget(id: string, dailyBudget: number) {
    return this.prisma.adCampaign.update({
      where: { id },
      data: { dailyBudget }
    })
  }

  // 获取计划的投放数据
  async getCampaignPerformance(id: string, dateRange?: { start: string; end: string }) {
    const where = {
      campaignId: id,
      ...(dateRange && {
        statDate: {
          gte: new Date(dateRange.start),
          lte: new Date(dateRange.end)
        }
      })
    }

    const data = await this.prisma.adPerformance.findMany({
      where,
      orderBy: { statDate: 'asc' }
    })

    // 汇总数据
    const summary = {
      totalImpression: 0,
      totalClick: 0,
      totalCost: 0,
      avgCTR: 0,
      avgCPC: 0
    }

    for (const item of data) {
      summary.totalImpression += item.impression
      summary.totalClick += item.click
      summary.totalCost += Number(item.cost)
    }

    if (summary.totalImpression > 0) {
      summary.avgCTR = (summary.totalClick / summary.totalImpression) * 100
    }
    if (summary.totalClick > 0) {
      summary.avgCPC = summary.totalCost / summary.totalClick
    }

    return {
      summary,
      daily: data
    }
  }
}
```

#### Step 3: 创建 Controller

创建 `server/src/modules/operation/ad-campaign.controller.ts`：

```typescript
import {
  Body,
  Controller,
  Delete,
  Get,
  Param,
  Post,
  Put,
  Query,
  Req,
  UseGuards
} from '@nestjs/common'
import { Request } from 'express'
import { JwtAuthGuard } from '../auth/jwt-auth.guard'
import { AdCampaignService } from './ad-campaign.service'

type AdminRequest = Request & { admin?: any }

@Controller('api/admin/ad/campaigns')
@UseGuards(JwtAuthGuard)
export class AdCampaignController {
  constructor(private readonly service: AdCampaignService) {}

  @Get()
  async listCampaigns(@Query() query: Record<string, unknown>) {
    return { data: await this.service.listCampaigns(query as any) }
  }

  @Post()
  async createCampaign(@Body() body: unknown, @Req() req: AdminRequest) {
    return { data: await this.service.createCampaign(body, req.admin?.username) }
  }

  @Get(':id')
  async getCampaignDetail(@Param('id') id: string) {
    return { data: await this.service.getCampaignDetail(id) }
  }

  @Put(':id/status')
  async updateStatus(@Param('id') id: string, @Body() body: any) {
    return { data: await this.service.updateCampaignStatus(id, body.status, body.rejectReason) }
  }

  @Put(':id/budget')
  async updateBudget(@Param('id') id: string, @Body() body: { dailyBudget: number }) {
    return { data: await this.service.updateDailyBudget(id, body.dailyBudget) }
  }

  @Get(':id/performance')
  async getPerformance(@Param('id') id: string, @Query() query: any) {
    return { data: await this.service.getCampaignPerformance(id, query.dateRange ? JSON.parse(query.dateRange) : undefined) }
  }
}
```

## 常见问题

### Q: 迁移期间会影响现有广告投放吗？
A: 不会。旧的 ads 表被重命名为 ad_creatives，但数据完整保留。用户端 API 继续调用旧逻辑，不受影响。

### Q: 如何回滚？
A: 如果出现问题，可以：
1. 回滚代码到上一个版本
2. 删除新表（ad_campaigns, ad_performance 等）
3. 将 ad_creatives 表改名回 ads
4. 恢复数据库备份

### Q: 旧数据如何迁移？
A: 见本文档"Phase 1: 数据库迁移"部分的数据迁移脚本。建议在非峰值时段执行。

### Q: 开发周期是多久？
A: 预计 5-7 个工作日
- 数据库迁移：1-2 天
- 后端开发：2-3 天
- 前端开发：2-3 天
- 测试和修复：1 天
